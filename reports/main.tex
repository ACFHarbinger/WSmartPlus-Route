        \textbf{ARP/GRP} & CARP, CGRP & Operational demand is distributed continuously along network edges rather than discrete nodes. \\

        \midrule

        Maximizing collected revenue under a strict time/distance budget & VRPP / TOP (Team Orienteering) \\

        Jointly optimizing delivery quantities and vehicle routes & IRP (Inventory Routing Problem) \\
        \midrule
        \textbf{Specialized Topologies} \\
        Demand is located on the streets/edges (e.g., snow plowing) & CARP (Capacitated Arc Routing) \\

        Human transit with strict maximum ride-time limits & DARP (Dial-a-Ride Problem) \\
        Simultaneous drop-off and pickup at the exact same stop & VRPSPD (VRP with Sim. Pickup/Delivery) \\
        Optimizing where to build depots alongside routing vehicles & LRP (Location Routing Problem) \\
        Urban distribution via intermediate satellite hubs & ME-LRP / 2E-VRP (Two-Echelon VRP) \\
        \midrule
        \textbf{Uncertainty \& Dynamism} \\
        Demands are unknown until arrival; requires recourse detours & SVRP (Stochastic VRP) \\
        Adversarial/worst-case demand bounding to guarantee zero stockouts & RVRP / RIRP (Robust VRP / IRP) \\
        New customer requests arrive continuously during execution & DVRP (Dynamic VRP) \\
        \bottomrule

\end{table}

\subsection*{Taxonomy of Neural Combinatorial Optimization Architectures}

\begin{table}[H]

    \caption{Classification of NCO models by construction strategy and mechanism.}
    \begin{tabularx}{\textwidth}{@{} l l X @{}}
        \toprule
        \textbf{Category} & \textbf{Key Architectures} & \textbf{Core Mechanism} \\
        \midrule
        \textbf{Autoregressive Constructive} & Ptr-Net, AM, POMO, SymNCO & Builds solutions token-by-token; decoder attends over the graph embedding at each step. \\
        \textbf{Non-Autoregressive} & NARGNN, DPDP & Outputs an edge-probability heatmap in a single pass; requires separate decoding (e.g., MCTS, greedy). \\
        \textbf{Improvement / Local Search} & L2I, N2S, NeuOpt, DACT & Learns to iteratively select and apply topological perturbations (e.g., $k$-opt) to existing feasible routes. \\
        \textbf{Meta-Heuristic Augmented} & DeepACO, DR-ALNS, GLOP & Replaces hand-crafted heuristic components in classical OR algorithms with trained neural networks. \\
        \textbf{Predict-Once} & GCN + Beam Search, DIFUSCO & Single forward pass yields guidance (heatmaps, candidate lists) that an independent classical solver consumes. \\
        \textbf{Predict-Many} & DPDP, Neural MCTS & Repeated neural queries within a search loop to dynamically score and guide tree expansion. \\
        \textbf{Multi-Task / Generalist} & MVMoE, Omni-VRP & A single set of weights handles multiple routing variants via sparse routing or conditional constraint tokens. \\
        \bottomrule
    \end{tabularx}
\end{table}

        \midrule

        \textbf{Risk Management} & Deadline-Driven & Projects exact days until guaranteed overflow using mean accumulation $\mu$. \\
        & CVaR / Wasserstein & Evaluates statistical tail risk or distributionally robust ambiguity balls to prevent stockouts under uncertainty. \\
        \textbf{Economic} & Spatial ROI & Normalizes expected node revenue by its marginal routing insertion cost ($r_i / 2d_{0,i}$). \\
        & Lagrangian Reduced-Cost & Evaluates LP relaxation shadow prices; mandates nodes with strictly positive marginal utility. \\
        \textbf{Spatial Synergy} & Clarke-Wright Savings & Flags nodes exhibiting minimum fill requirements and positive spatial savings ($s_{ij} > 0$) with critical nodes. \\
        \textbf{Simulation} & Whittle Index & RMAB reinforcement learning formulation computing the exact subsidy for passivity via Value Iteration. \\
        \textbf{Dispatchers} & Contextual Thompson & Multi-Armed Bandit dynamically selecting the optimal filtering strategy based on current environment state. \\
        \bottomrule
    \end{tabularx}
\end{table}

\subsection*{Combinatorial Search Operator Taxonomy}

\begin{table}[H]
    \centering
    \caption{Low-level combinatorial operators defining the atomic mutations of routing graphs.}
    \begin{tabularx}{\textwidth}{@{} l l X @{}}
        \toprule
        \textbf{Operator Class} & \textbf{Notable Operators} & \textbf{Structural Impact} \\
        \midrule
        \textbf{Crossover} & OX, PMX, GPX, ERX & Inherits absolute positions, relative sequences, or spatial partitions from two or more parent solutions. \\
        \textbf{Destroy (Ruin)} & Worst, Shaw, Cluster, String & Ejects specific nodes based on marginal cost, spatial/demand relatedness, or geometric contiguousness. \\
        \textbf{Mutation} & Swap, Inversion, Scramble & Introduces localized stochastic variations (e.g., reversing substrings) to maintain genetic diversity. \\
        \textbf{Generalized (GENI)} & Unstringing / Stringing & Evaluates multi-arc, non-adjacent vertex insertions and deletions to bypass standard topological adjacency bounds. \\
        \textbf{Perturbation} & Double Bridge, Kick & Violently fragments and reconnects quarters of a tour to ensure absolute escape from local optima basins. \\
        \textbf{Recreate (Repair)} & Greedy, Deep, Farthest & Restores graph feasibility by inserting nodes to minimize cost or maximize geometric coverage. \\
        \bottomrule
    \end{tabularx}
\end{table}

\subsection*{Acceptance Criteria in Trajectory-Based Meta-heuristics}

\begin{table}[H]

    \caption{Mathematical rules governing trajectory transitions and objective deterioration tolerance.}
    \begin{tabularx}{\textwidth}{@{} l l l X @{}}
        \toprule
        \textbf{Category} & \textbf{Criterion} & \textbf{Type} & \textbf{Acceptance Rule / Feature} \\
        \midrule
        \textbf{Elitist} & Only Improving (OI) & Deterministic & $\Delta f < 0$ (Strict greedy descent). \\
        & Improving \& Equal (IE) & Deterministic & $\Delta f \le 0$ (Permits random walks across neutral plateaus). \\
        \textbf{Thermodynamic} & Boltzmann (BMC) & Stochastic & Accepts worsening moves with probability $\exp(-\Delta f / T)$. \\
        & Cauchy SA & Stochastic & Utilizes heavier-tailed probability for escaping deep, wide basins. \\
        \textbf{Bounded} & Great Deluge (GD) & Deterministic & Cost must remain below a monotonically dropping water level $W$. \\
        & Record-to-Record & Deterministic & Cost must be within margin $\delta$ of the global best ($f(s^*) + \delta$). \\
        \textbf{Memory-Based} & Late Acceptance (LAHC) & Deterministic & Compares candidate against the cost from exactly $L$ steps ago. \\
        & Old Bachelor (OBA) & Adaptive & Threshold tightens on success and relaxes on consecutive rejections. \\
        \textbf{Multi-Objective} & Pareto Dominance (PD) & Deterministic & Candidate must strictly dominate the incumbent across all objectives. \\
        \bottomrule
    \end{tabularx}
\end{table}
